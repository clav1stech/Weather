===== Fichier : Convert.py =====

import os
import glob
import re
import datetime
import pandas as pd
from functools import reduce
from openpyxl import Workbook
from openpyxl.styles import PatternFill, Font, Alignment, Border, Side
from openpyxl.utils.dataframe import dataframe_to_rows
from openpyxl.utils import get_column_letter

# --- Configuration ---
DOSSIER_ENTREE = "Anciens_Fichiers"
DOSSIER_SORTIE = "Forecasts"
os.makedirs(DOSSIER_SORTIE, exist_ok=True)

# --- Fonctions Utilitaires ---
def parse_old_filename(filepath):
    """ Extrait la date et le run depuis l'ancien format """
    match = re.search(r'prev-(\d{8})-(0Z|12Z)\.xlsx', os.path.basename(filepath))
    if match:
        dt = datetime.datetime.strptime(match.group(1), "%d%m%Y")
        run = match.group(2)
        heures = 12 if run == "12Z" else 0
        return dt + datetime.timedelta(hours=heures), match.group(1), run
    return datetime.datetime.min, None, None

def appliquer_style_tableau(ws, start_row):
    header_fill = PatternFill(start_color="34495E", end_color="34495E", fill_type="solid")
    header_font = Font(color="FFFFFF", bold=True)
    center_align = Alignment(horizontal="center", vertical="center")
    thin_border = Border(left=Side(style='thin'), right=Side(style='thin'), top=Side(style='thin'), bottom=Side(style='thin'))

    for col in range(1, ws.max_column + 1):
        ws.column_dimensions[get_column_letter(col)].width = 15 if col == 1 else 11

    for row in ws.iter_rows(min_row=start_row, max_row=ws.max_row, min_col=1, max_col=ws.max_column):
        for cell in row:
            cell.alignment = center_align
            cell.border = thin_border
            if cell.row == start_row:
                cell.fill = header_fill
                cell.font = header_font

# --- Traitement ---
fichiers_a_convertir = glob.glob(os.path.join(DOSSIER_ENTREE, "prev-*.xlsx"))
fichiers_a_convertir.sort(key=lambda x: parse_old_filename(x)[0])

for f_path in fichiers_a_convertir:
    current_dt, date_str, run_str = parse_old_filename(f_path)
    if not date_str:
        continue
        
    print(f"Conversion de : {os.path.basename(f_path)}...")
    nouveau_nom = f"Forecast-{date_str}-{run_str}.xlsx"
    chemin_sortie = os.path.join(DOSSIER_SORTIE, nouveau_nom)
    
    wb = Workbook()
    wb.remove(wb.active)
    model_dataframes = {}
    
    try:
        xls = pd.ExcelFile(f_path)
        # Détection souple des onglets
        for sheet_in_file in xls.sheet_names:
            new_sheet = None
            sheet_upper = sheet_in_file.upper()
            if "ECMWF" in sheet_upper: new_sheet = "ECMWF"
            elif "AIFS" in sheet_upper: new_sheet = "AIFS"
            elif "GEFS" in sheet_upper: new_sheet = "GEFS"
            
            if new_sheet:
                df = pd.read_excel(xls, sheet_name=sheet_in_file)
                model_dataframes[new_sheet] = df
                
                ws_data = wb.create_sheet(title=new_sheet)
                ws_data["A1"], ws_data["A2"] = f"Modèle : {new_sheet}", f"Informations : Récupéré depuis archive"
                ws_data["A1"].font = ws_data["A2"].font = Font(bold=True)
                
                for r_idx, row in enumerate(dataframe_to_rows(df, index=False, header=True), start=4):
                    for c_idx, value in enumerate(row, start=1):
                        ws_data.cell(row=r_idx, column=c_idx, value=value)
                appliquer_style_tableau(ws_data, start_row=4)
                
    except Exception as e:
        print(f"❌ Erreur lors de la lecture de {f_path}: {e}")
        continue

    # Sécurité : si aucun onglet n'a été récupéré, on passe au fichier suivant
    if not model_dataframes:
        print(f"⚠️ Aucun onglet modèle reconnu dans {f_path}. Fichier ignoré.")
        continue

    # --- Synthèse Multi-Modèles ---
    dfs_to_merge, all_members = [], []
    for mod_nom, df_mod in model_dataframes.items():
        membres = [c for c in df_mod.columns if c not in ['Date', 'Ech.']]
        df_temp = df_mod[['Date', 'Ech.'] + membres].copy()
        df_temp.columns = ['Date', 'Ech.'] + [f"{mod_nom}_{c}" for c in membres]
        all_members.extend([f"{mod_nom}_{c}" for c in membres])
        dfs_to_merge.append(df_temp)

    if dfs_to_merge:
        df_super = reduce(lambda left, right: pd.merge(left, right, on=['Date', 'Ech.'], how='outer'), dfs_to_merge)
        df_super = df_super[df_super['Date'].astype(str).str.contains(' 12Z', na=False)].copy()
        
        df_super['Date'] = df_super['Date'].astype(str).str.split(' ').str[0]
        # On conserve un objet Date natif
        df_super['Date'] = pd.to_datetime(df_super['Date'], errors='coerce').dt.date
        
        df_super['Ech_num'] = pd.to_numeric(df_super['Ech.'], errors='coerce')
        df_super = df_super.sort_values('Ech_num').drop(columns=['Ech_num']).reset_index(drop=True)
        
        m = df_super[all_members]
        m = m.apply(pd.to_numeric, errors='coerce')
        
        df_stats = pd.DataFrame()
        df_stats['Date'] = df_super['Date']
        df_stats['Ech'] = df_super['Ech.']
        df_stats['Min'] = m.min(axis=1).round(1)
        df_stats['P10'] = m.quantile(0.10, axis=1).round(1)
        df_stats['P25'] = m.quantile(0.25, axis=1).round(1)
        df_stats['Médiane'] = m.median(axis=1).round(1)
        df_stats['P75'] = m.quantile(0.75, axis=1).round(1)
        df_stats['P90'] = m.quantile(0.90, axis=1).round(1)
        df_stats['Max'] = m.max(axis=1).round(1)
        df_stats['Spread'] = (df_stats['P90'] - df_stats['P10']).round(1)
        df_stats['Ecart-type'] = m.std(axis=1).round(2)
        df_stats['Proba > 20°'] = ((m > 20).sum(axis=1) / m.notna().sum(axis=1)).fillna(0)

        # Calcul Historique et Deltas
        fichiers_existants = glob.glob(os.path.join(DOSSIER_SORTIE, "Forecast-*.xlsx"))
        fichiers_existants.sort(key=lambda x: datetime.datetime.strptime(re.search(r'Forecast-(\d{8})', os.path.basename(x)).group(1), "%d%m%Y"), reverse=True)
        
        limite_dt = current_dt - datetime.timedelta(days=5)
        
        for prev_file in fichiers_existants[:10]:
            match_prev = re.search(r'Forecast-(\d{8})-(0Z|12Z)\.xlsx', os.path.basename(prev_file))
            if not match_prev: continue
            
            prev_dt = datetime.datetime.strptime(match_prev.group(1), "%d%m%Y")
            prev_run = match_prev.group(2)
            prev_dt_full = prev_dt + datetime.timedelta(hours=12 if prev_run == "12Z" else 0)
            
            if limite_dt <= prev_dt_full < current_dt:
                try:
                    df_prev = pd.read_excel(prev_file, sheet_name="Synthèse", skiprows=4)
                    if 'Médiane' in df_prev.columns:
                        # On s'assure que la colonne lue est traitée comme une date pour que la jointure fonctionne
                        df_prev['Date'] = pd.to_datetime(df_prev['Date'], errors='coerce').dt.date
                        merged = df_stats.merge(df_prev[['Date', 'Médiane']], on='Date', how='left', suffixes=('', '_Prev'))
                        col_delta = f"Δ {prev_dt.strftime('%d/%m')} {prev_run}"
                        df_stats[col_delta] = (df_stats['Médiane'] - merged['Médiane_Prev']).round(1)
                except Exception:
                    pass

        ws_synth = wb.create_sheet(title="Synthèse", index=0)
        ws_synth["A1"] = "Synthèse Multi-Modèles (Super-Ensemble) - Echéances 12Z"
        ws_synth["A2"] = f"Généré depuis l'archive le : {datetime.datetime.now().strftime('%d/%m/%Y %H:%M')}"
        ws_synth["A3"] = f"Runs convertis pour : {date_str} {run_str}"
        ws_synth["A1"].font = ws_synth["A2"].font = ws_synth["A3"].font = Font(bold=True)
        
        for r_idx, row in enumerate(dataframe_to_rows(df_stats, index=False, header=True), start=5):
            for c_idx, value in enumerate(row, start=1):
                cell = ws_synth.cell(row=r_idx, column=c_idx, value=value)
                col_name = df_stats.columns[c_idx-1]
                
                # Formatage natif
                if col_name == 'Date' and r_idx > 5:
                    cell.number_format = 'dd-mmm'
                    
                if col_name == 'Proba > 20°' and r_idx > 5:
                    cell.number_format = '0.0%'
                    
                if str(col_name).startswith('Δ') and r_idx > 5 and pd.notna(value):
                    if value > 0:
                        cell.font = Font(color="D32F2F", bold=True)
                    elif value < 0:
                        cell.font = Font(color="1976D2", bold=True)
                        
        appliquer_style_tableau(ws_synth, start_row=5)
        ws_synth.freeze_panes = "C6"

    wb.save(chemin_sortie)
    print(f"✅ Fichier sauvegardé : {nouveau_nom}")

print("\n🚀 Conversion totale terminée !")

===== Fichier : Forecast.py =====

import os
import sys
import glob
import datetime
import re
import requests
import pandas as pd
import numpy as np
from bs4 import BeautifulSoup
from functools import reduce
from collections import Counter
from openpyxl import Workbook
from openpyxl.styles import PatternFill, Font, Alignment, Border, Side
from openpyxl.utils.dataframe import dataframe_to_rows
from openpyxl.utils import get_column_letter

# --- Configuration ---
DOSSIER_SORTIE = "Forecasts"
os.makedirs(DOSSIER_SORTIE, exist_ok=True)

# Switch du run souhaité : "0Z" ou "12Z"
# Priorité : argument CLI (ex: `python Forecast.py 12Z`), puis variable d'env, sinon "0Z".
_run_arg = sys.argv[1] if len(sys.argv) > 1 and sys.argv[1] in ("0Z", "12Z") else None
CHOIX_RUN = _run_arg or os.environ.get("FORECAST_RUN", "0Z")
run_param = "0" if CHOIX_RUN == "0Z" else "12"
heure_attendue = 12 if CHOIX_RUN == "12Z" else 0

modeles = [
    {"nom": "ECMWF ENS", "sheet": "ECMWF", "url": f"https://www.meteociel.fr/modeles/ecmwfens_table.php?ext=1&x=&lat=48.8621&lon=2.33936&ville=Paris&run={run_param}"},
    {"nom": "AIFS ENS", "sheet": "AIFS", "url": f"https://www.meteociel.fr/modeles/ecmwfens_table.php?ext=1&x=&lat=48.8621&lon=2.33936&ville=Paris&aifs=1&run={run_param}"},
    {"nom": "GEFS", "sheet": "GEFS", "url": f"https://www.meteociel.fr/modeles/gefs_table.php?ext=1&x=&lat=48.8621&lon=2.33936&ville=Paris&run={run_param}"}
]

headers = {"User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64)"}

# --- Fonctions Utilitaires ---
def parse_file_datetime(filepath):
    match = re.search(r'Forecast-(\d{8})(?:-(0Z|12Z))?\.xlsx', os.path.basename(filepath))
    if match:
        dt = datetime.datetime.strptime(match.group(1), "%d%m%Y")
        run = match.group(2) if match.group(2) else "12Z" 
        heures = 12 if run == "12Z" else 0
        return dt + datetime.timedelta(hours=heures)
    return datetime.datetime.min

def parser_tableau_meteociel(html):
    soup = BeautifulSoup(html, 'html.parser')
    table = soup.find('table', class_='gefs')
    if not table: return None, None
    
    data, colors = [], []
    for tr in table.find_all('tr'):
        row_data, row_colors = [], []
        for td in tr.find_all(['td', 'th']):
            row_data.append(td.get_text(strip=True))
            bg = td.get('bgcolor')
            row_colors.append(bg.lstrip('#') if bg else None)
        data.append(row_data)
        colors.append(row_colors)
        
    df = pd.DataFrame(data[1:], columns=data[0])
    for col in df.columns:
        if col not in ['Date', 'Ech.']:
            df[col] = pd.to_numeric(df[col], errors='coerce')
            
    return df, colors[1:]

def appliquer_style_tableau(ws, start_row, style_type="standard", df_cols=None):
    center_align = Alignment(horizontal="center", vertical="center")
    thin_border = Border(left=Side(style='thin'), right=Side(style='thin'), top=Side(style='thin'), bottom=Side(style='thin'))
    font_white = Font(color="FFFFFF", bold=True)
    
    default_fill = PatternFill(start_color="34495E", end_color="34495E", fill_type="solid")
    color_ecmwf = PatternFill(start_color="1F618D", end_color="1F618D", fill_type="solid") 
    color_aifs = PatternFill(start_color="1E8449", end_color="1E8449", fill_type="solid")  
    color_gefs = PatternFill(start_color="B9770E", end_color="B9770E", fill_type="solid")  
    color_div = PatternFill(start_color="7D3C98", end_color="7D3C98", fill_type="solid")   

    for col in range(1, ws.max_column + 1):
        col_letter = get_column_letter(col)
        ws.column_dimensions[col_letter].width = 15 if col == 1 else 11
        
        header_fill = default_fill
        if df_cols:
            col_name = df_cols[col - 1]
            if "ECMWF" in col_name: header_fill = color_ecmwf
            elif "AIFS" in col_name: header_fill = color_aifs
            elif "GEFS" in col_name: header_fill = color_gefs
            elif style_type == "compare" and "Divergence" in col_name: header_fill = color_div

        for row in ws.iter_rows(min_row=start_row, max_row=ws.max_row, min_col=col, max_col=col):
            cell = row[0]
            cell.alignment = center_align
            cell.border = thin_border
            if cell.row == start_row:
                cell.fill = header_fill
                cell.font = font_white

# --- Initialisation ---
wb = Workbook()
wb.remove(wb.active)

model_dataframes = {}
model_colors = {}
runs_info = []

# --- 1. Extraction Tolérante ---
extractions_brutes = []

for mod in modeles:
    print(f"Téléchargement de {mod['nom']}...")
    try:
        response = requests.get(mod["url"], headers=headers, timeout=10)
        response.encoding = 'windows-1252'
        html = response.text
        
        match = re.search(r'(Run .*?(\d{2}/\d{2}/\d{4})\s+(\d{1,2})[Zz])', html)
        
        if not match:
            print(f"⚠️ AVERTISSEMENT : Impossible de lire la date pour {mod['nom']} (Tableau non publié ou format cassé).")
            continue
            
        run_info_complet = match.group(1)
        run_date_ext = match.group(2)
        run_heure_ext = int(match.group(3))
        
        df, row_colors = parser_tableau_meteociel(html)
        if df is not None:
            extractions_brutes.append({
                'nom': mod['nom'],
                'sheet': mod['sheet'],
                'df': df,
                'colors': row_colors,
                'date': run_date_ext,
                'heure': run_heure_ext,
                'info': run_info_complet
            })
    except Exception as e:
        print(f"⚠️ AVERTISSEMENT : Échec de la connexion pour {mod['nom']} ({e}).")

if not extractions_brutes:
    sys.exit("\n❌ ERREUR CRITIQUE : Absolument aucun modèle n'a pu être récupéré. Exécution annulée.")

# --- 2. Tri et Vote Majoritaire (Gestion de la désynchronisation) ---
# Étape A : On s'assure d'abord que les modèles ont bien respecté l'heure demandée via l'URL
modeles_heure_valide = []
for m in extractions_brutes:
    if m['heure'] == heure_attendue:
        modeles_heure_valide.append(m)
    else:
        print(f"  ❌ {m['nom']} écarté (Mauvais run : {m['heure']}Z au lieu de {heure_attendue}Z)")

if not modeles_heure_valide:
    sys.exit(f"\n❌ ERREUR : Aucun modèle n'est disponible pour le run {CHOIX_RUN}.")

# Étape B : Vote majoritaire sur la DATE du jour (pour isoler les modèles en retard)
dates = [m['date'] for m in modeles_heure_valide]
counts = Counter(dates)

# En cas d'égalité stricte (ex: 1 modèle à J, 1 modèle à J-1), on privilégie la date la plus récente.
def rank_date(item):
    date_str, count = item
    return (count, datetime.datetime.strptime(date_str, "%d/%m/%Y"))

majority_date = sorted(counts.items(), key=rank_date, reverse=True)[0][0]

date_file_str = datetime.datetime.strptime(majority_date, "%d/%m/%Y").strftime("%d%m%Y")
fichier_actuel = os.path.join(DOSSIER_SORTIE, f"Forecast-{date_file_str}-{CHOIX_RUN}.xlsx")

print(f"\n👉 Date majoritaire retenue : {majority_date} (Run {CHOIX_RUN})")

for m in modeles_heure_valide:
    if m['date'] == majority_date:
        print(f"  ✅ {m['nom']} intégré au super-ensemble.")
        model_dataframes[m['nom']] = m['df']
        model_colors[m['nom']] = m['colors']
        runs_info.append(f"{m['nom']} ({m['info']})")
        
        # Création de sa feuille individuelle
        ws_data = wb.create_sheet(title=m["sheet"])
        ws_data["A1"], ws_data["A2"] = f"Modèle : {m['nom']}", f"Informations : {m['info']}"
        ws_data["A1"].font = ws_data["A2"].font = Font(bold=True)
        
        for r_idx, row in enumerate(dataframe_to_rows(m['df'], index=False, header=True), start=4):
            for c_idx, value in enumerate(row, start=1):
                cell = ws_data.cell(row=r_idx, column=c_idx, value=value)
                if r_idx > 4: 
                    color_hex = m['colors'][r_idx - 5][c_idx - 1]
                    if color_hex:
                        cell.fill = PatternFill(start_color=color_hex, end_color=color_hex, fill_type='solid')
        appliquer_style_tableau(ws_data, start_row=4)
    else:
        print(f"  ❌ {m['nom']} écarté (En retard : toujours bloqué à la date du {m['date']})")

# --- 3. Définition de l'historique ---
fichiers_existants = glob.glob(os.path.join(DOSSIER_SORTIE, "Forecast-*.xlsx"))
fichiers_tries = sorted(fichiers_existants, key=parse_file_datetime, reverse=True)

current_dt = parse_file_datetime(fichier_actuel)
limite_dt = current_dt - datetime.timedelta(days=5)

fichiers_precedents = [
    f for f in fichiers_tries 
    if parse_file_datetime(f) < current_dt and parse_file_datetime(f) >= limite_dt
][:10]

# --- 4. Synthèses ---
print("\nCalcul des statistiques de synthèse...")

dfs_to_merge, all_members = [], []
dfs_compare_list = []
det_col_names = {}  # short_nom -> nom de colonne renommée dans df_super

for mod_nom, df_mod in model_dataframes.items():
    short_nom = mod_nom.replace(" ENS", "")
    has_det = 'DET' in df_mod.columns
    membres = [c for c in df_mod.columns if c not in ['Date', 'Ech.', 'DET']]

    extra_cols = ['DET'] if has_det else []
    df_temp = df_mod[['Date', 'Ech.'] + membres + extra_cols].copy()
    new_cols = [f"{short_nom}_{c}" for c in membres]
    det_renamed = [f"{short_nom}_DET"] if has_det else []
    df_temp.columns = ['Date', 'Ech.'] + new_cols + det_renamed
    all_members.extend(new_cols)
    if has_det:
        det_col_names[short_nom] = f"{short_nom}_DET"
    dfs_to_merge.append(df_temp)

    m_indiv = df_mod[membres].apply(pd.to_numeric, errors='coerce')
    df_mod_stats = pd.DataFrame()
    df_mod_stats['Date'] = df_mod['Date']
    df_mod_stats['Ech'] = df_mod['Ech.']
    df_mod_stats[f'{short_nom} P10'] = m_indiv.quantile(0.10, axis=1).round(1)
    df_mod_stats[f'{short_nom} Médiane'] = m_indiv.median(axis=1).round(1)
    if has_det:
        df_mod_stats[f'{short_nom} DET'] = pd.to_numeric(df_mod['DET'], errors='coerce').round(1)
    df_mod_stats[f'{short_nom} P90'] = m_indiv.quantile(0.90, axis=1).round(1)
    df_mod_stats[f'{short_nom} Spread'] = (df_mod_stats[f'{short_nom} P90'] - df_mod_stats[f'{short_nom} P10']).round(1)
    dfs_compare_list.append(df_mod_stats)

if dfs_to_merge:
    # --- Super-Ensemble ---
    df_super = reduce(lambda left, right: pd.merge(left, right, on=['Date', 'Ech.'], how='outer'), dfs_to_merge)
    df_super = df_super[df_super['Date'].str.contains(' 12Z', na=False)].copy()
    
    df_super['Date'] = df_super['Date'].str.split(' ').str[0]
    df_super['Date'] = pd.to_datetime(df_super['Date'], errors='coerce').dt.date
    
    df_super['Ech_num'] = pd.to_numeric(df_super['Ech.'], errors='coerce')
    df_super = df_super.sort_values('Ech_num').drop(columns=['Ech_num']).reset_index(drop=True)
    
    m = df_super[all_members]
    
    df_stats = pd.DataFrame()
    df_stats['Date'] = df_super['Date']
    df_stats['Ech'] = df_super['Ech.']
    df_stats['Min'] = m.min(axis=1).round(1)
    df_stats['P10'] = m.quantile(0.10, axis=1).round(1)
    df_stats['P25'] = m.quantile(0.25, axis=1).round(1)
    df_stats['Médiane'] = m.median(axis=1).round(1)
    for short_nom, det_col in det_col_names.items():
        if det_col in df_super.columns:
            df_stats[f'{short_nom} DET'] = pd.to_numeric(df_super[det_col], errors='coerce').round(1)
    df_stats['P75'] = m.quantile(0.75, axis=1).round(1)
    df_stats['P90'] = m.quantile(0.90, axis=1).round(1)
    df_stats['Max'] = m.max(axis=1).round(1)
    df_stats['Spread'] = (df_stats['P90'] - df_stats['P10']).round(1)
    df_stats['Ecart-type'] = m.std(axis=1).round(2)
    df_stats['Proba > 20°'] = ((m > 20).sum(axis=1) / m.notna().sum(axis=1)).fillna(0)
    
    df_stats_ech_num = pd.to_numeric(df_stats['Ech'], errors='coerce')
    
    # --- Deltas ---
    for prev_file in fichiers_precedents:
        try:
            prev_dt = parse_file_datetime(prev_file)
            delta_hours = int((current_dt - prev_dt).total_seconds() / 3600)
            
            df_prev = pd.read_excel(prev_file, sheet_name="Synthèse", skiprows=4)
            if 'Médiane' in df_prev.columns and 'Ech' in df_prev.columns:
                
                df_prev['Ech_Num'] = pd.to_numeric(df_prev['Ech'], errors='coerce')
                prev_med_map = df_prev.set_index('Ech_Num')['Médiane'].to_dict()
                
                heure_str = "12Z" if prev_dt.hour == 12 else "0Z"
                col_delta = f"Δ {prev_dt.strftime('%d-%b')} {heure_str}"
                
                mediane_prev_aligned = df_stats_ech_num.apply(lambda e: prev_med_map.get(e + delta_hours, np.nan))
                df_stats[col_delta] = (df_stats['Médiane'] - mediane_prev_aligned).round(1)
        except Exception:

def parse_file_datetime(filepath):
    import re
    match = re.search(r'Forecast-(\d{8})-(0Z|12Z)\.xlsx', os.path.basename(filepath))
    if match:
        dt = datetime.datetime.strptime(match.group(1), "%d%m%Y")
        run = match.group(2)
        heures = 12 if run == "12Z" else 0
        return dt + datetime.timedelta(hours=heures)
    return datetime.datetime.min

def formater_date_meteociel(dt):
    jours = ["Lun", "Mar", "Mer", "Jeu", "Ven", "Sam", "Dim"]
    return f"{jours[dt.weekday()]} {dt.day:02d} {dt.hour:02d}Z"

def generer_couleur_t850(temp):
    if pd.isna(temp): return None
    if temp < -10: return "0D47A1"
    elif temp < -5: return "1976D2"
    elif temp < 0: return "64B5F6"
    elif temp < 5: return "81C784"
    elif temp < 10: return "DCE775"
    elif temp < 15: return "FFF176"
    elif temp < 20: return "FFB74D"
    elif temp < 25: return "FF7043"
    else: return "D32F2F"

class WeatherAPIClient:
    def __init__(self, lat, lon):
        self.lat = lat
        self.lon = lon
        self.lon_360 = lon if lon >= 0 else 360 + lon

class GEFSFetcher(WeatherAPIClient):
    def fetch_t850(self, run_dt):
        date_str = run_dt.strftime("%Y%m%d")
        hour_str = f"{run_dt.hour:02d}"
        url = f"https://nomads.ncep.noaa.gov/dods/gefs/gefs{date_str}/gefs_pgrb2ap5_all_{hour_str}z"
        
        try:
            ds = xr.open_dataset(url)
            ds_point = ds['tmpprs'].sel(lon=self.lon_360, lat=self.lat, lev=850, method='nearest')
            
            df_raw = ds_point.to_dataframe().reset_index()
            df_raw['tmpprs'] = df_raw['tmpprs'] - 273.15 
            df_raw['Ech.'] = ((df_raw['time'] - run_dt).dt.total_seconds() / 3600).astype(int)
            df_raw['Date'] = df_raw['time'].apply(formater_date_meteociel)
            
            df_pivot = df_raw.pivot(index=['time', 'Date', 'Ech.'], columns='ens', values='tmpprs').reset_index()
            df_pivot = df_pivot.sort_values('time').drop(columns=['time'])
            df_pivot.columns = [str(int(c)) if isinstance(c, (int, float)) else str(c) for c in df_pivot.columns]
            
            colors_matrix = []
            for _, row in df_pivot.drop(columns=['Date', 'Ech.']).iterrows():
                colors_matrix.append([generer_couleur_t850(val) for val in row])
                
            return df_pivot, colors_matrix, f"Run GEFS {date_str} {hour_str}Z"
        except Exception as e:
            return None, None, str(e)

class ECMWFFetcher(WeatherAPIClient):
    def fetch_t850(self, run_dt, model="ifs"):
        if not ECMWFClient: return None, None, "ecmwf-opendata non installé"
        
        client = ECMWFClient(source="ecmwf", email=ECMWF_API_EMAIL, key=ECMWF_API_KEY)
        target_file = f"temp_{model}_{run_dt.strftime('%Y%m%d%H')}_t850.grib"
        
        try:
            client.retrieve(
                date=run_dt.strftime("%Y%m%d"),
                time=run_dt.hour,
                model=model,
                type="ep",
                param="t",
                levelist=850,
                target=target_file
            )
            
            ds = xr.open_dataset(target_file, engine='cfgrib')
            ds_point = ds['t'].sel(longitude=self.lon, latitude=self.lat, method='nearest')
            
            df_raw = ds_point.to_dataframe().reset_index()
            df_raw['t'] = df_raw['t'] - 273.15
            df_raw['time'] = pd.to_datetime(df_raw['time'])
            df_raw['valid_time'] = df_raw['time'] + df_raw['step']
            df_raw['Ech.'] = (df_raw['step'].dt.total_seconds() / 3600).astype(int)
            df_raw['Date'] = df_raw['valid_time'].apply(formater_date_meteociel)
            
            df_pivot = df_raw.pivot(index=['valid_time', 'Date', 'Ech.'], columns='number', values='t').reset_index()
            df_pivot = df_pivot.sort_values('valid_time').drop(columns=['valid_time'])
            df_pivot.columns = [str(c) for c in df_pivot.columns]
            
            colors_matrix = []
            for _, row in df_pivot.drop(columns=['Date', 'Ech.']).iterrows():
                colors_matrix.append([generer_couleur_t850(val) for val in row])
                
            os.remove(target_file)
            return df_pivot, colors_matrix, f"Run ECMWF {model.upper()} {run_dt.strftime('%d/%m/%Y %H')}Z"
        except Exception as e:
            if os.path.exists(target_file): os.remove(target_file)
            return None, None, str(e)

def formater_date_meteociel(dt):
    jours = ["Lun", "Mar", "Mer", "Jeu", "Ven", "Sam", "Dim"]
    return f"{jours[dt.weekday()]} {dt.day:02d} {dt.hour:02d}Z"

def generer_couleur_t850(temp):
    if pd.isna(temp): return None
    if temp < -10: return "0D47A1"
    elif temp < -5: return "1976D2"
    elif temp < 0:  return "64B5F6"
    elif temp < 5:  return "81C784"
    elif temp < 10: return "DCE775"
    elif temp < 15: return "FFF176"
    elif temp < 20: return "FFB74D"
    elif temp < 25: return "FF7043"
    else:           return "D32F2F"


class WeatherAPIClient:
    def __init__(self, lat, lon):
        self.lat = lat
        self.lon = lon
        self.lon_360 = lon if lon >= 0 else 360 + lon


# ==============================================================================
# GEFS via AWS S3 (remplace l'accès OpenDAP NOAA, retiré fin 2024)
# Bucket : s3://noaa-gefs-pds  —  accès anonyme, région us-east-1
# Structure : gefs.YYYYMMDD/HHz/atmos/pgrb2ap5/gepNN.tHHz.pgrb2a.p5.fFFF
#   gec00  = membre de contrôle
#   gep01..gep30 = membres perturbés
# ==============================================================================
class GEFSFetcher(WeatherAPIClient):

    # Membres standard de l'ensemble GEFS (contrôle + 30 perturbés)
    MEMBRES = ["c00"] + [f"p{i:02d}" for i in range(1, 31)]
    MEMBRES_PREFIXE = {"c00": "gec00", **{f"p{i:02d}": f"gep{i:02d}" for i in range(1, 31)}}

    def _build_s3_path(self, run_dt: datetime.datetime, membre: str, echeance_h: int) -> str:
        date_str  = run_dt.strftime("%Y%m%d")
        hour_str  = f"{run_dt.hour:02d}"
        prefix    = self.MEMBRES_PREFIXE[membre]
        ech_str   = f"{echeance_h:03d}"
        return (
            f"noaa-gefs-pds/gefs.{date_str}/{hour_str}z/atmos/pgrb2ap5/"
            f"{prefix}.t{hour_str}z.pgrb2a.p5.f{ech_str}"
        )

    def _get_fs(self) -> "s3fs.S3FileSystem":
        if s3fs is None:
            raise ImportError("s3fs requis : pip install s3fs")
        return s3fs.S3FileSystem(
            anon=True,
            client_kwargs={"region_name": "us-east-1"},
        )

    def _lire_t850_membre(
        self,
        fs: "s3fs.S3FileSystem",
        run_dt: datetime.datetime,
        membre: str,
        echeances: list[int],
    ) -> pd.Series:
        """Retourne une Series indexée par échéance (h) avec T850 en °C."""
        valeurs = {}
        for ech in echeances:
            path = self._build_s3_path(run_dt, membre, ech)
            try:
                with fs.open(path, "rb") as f:
                    raw = f.read()
                with tempfile.NamedTemporaryFile(suffix=".grib2", delete=False) as tmp:
                    tmp.write(raw)
                    tmp_path = tmp.name
                try:
                    # filtre cfgrib : température sur niveaux de pression isobariques
                    ds = xr.open_dataset(
                        tmp_path,
                        engine="cfgrib",
                        backend_kwargs={
                            "filter_by_keys": {
                                "typeOfLevel": "isobaricInhPa",
                                "shortName": "t",
                                "level": 850,
                            }
                        },
                    )
                    # Sélection du point le plus proche (grille 0.5°)
                    t_val = float(
                        ds["t"]
                        .sel(
                            longitude=self.lon_360,
                            latitude=self.lat,
                            method="nearest",
                        )
                        .values
                    ) - 273.15  # Kelvin → Celsius
                    valeurs[ech] = round(t_val, 1)
                finally:
                    os.remove(tmp_path)
            except s3fs.core.S3FileNotFoundError:
                # Fichier absent : échéance non encore disponible ou run incomplet
                valeurs[ech] = np.nan
            except PermissionError as exc:
                raise PermissionError(
                    f"Accès S3 refusé — vérifier la disponibilité du bucket NOAA "
                    f"(noaa-gefs-pds). Détail : {exc}"
                ) from exc
            except Exception:
                valeurs[ech] = np.nan
        return pd.Series(valeurs, name=membre)

    def fetch_t850(self, run_dt: datetime.datetime):
        """
        Retourne (df_pivot, colors_matrix, info_str) ou (None, None, message_erreur).
        df_pivot : colonnes ['Date', 'Ech.', 'c00', 'p01', ..., 'p30']
        """
        if xr is None:
            return None, None, "xarray non installé"

        # Échéances GEFS pgrb2ap5 : toutes les 3 h jusqu'à H+240, puis 6 h
        echeances = list(range(0, 243, 3)) + list(range(246, 387, 6))

        try:
            fs = self._get_fs()
        except (ImportError, Exception) as exc:
            return None, None, str(exc)

        records = {}  # {ech: {membre: valeur}}
        erreurs_s3 = []

        for membre in self.MEMBRES:
            try:
                serie = self._lire_t850_membre(fs, run_dt, membre, echeances)
                for ech, val in serie.items():
                    records.setdefault(ech, {})["Ech."] = ech
                    records[ech][membre] = val
            except PermissionError as exc:
                # Erreur d'accès globale : inutile de continuer
                return None, None, str(exc)
            except Exception as exc:
                erreurs_s3.append(f"{membre}: {exc}")

        if not records:
            detail = " | ".join(erreurs_s3[:3]) if erreurs_s3 else "Aucune donnée"
            return None, None, f"GEFS S3 — aucune donnée récupérée. {detail}"

        df = pd.DataFrame.from_dict(records, orient="index").sort_index()
        df["Ech."] = df["Ech."].astype(int)

        # Colonne Date (échéance → datetime UTC → label Météociel)
        df["Date"] = df["Ech."].apply(
            lambda h: formater_date_meteociel(run_dt + datetime.timedelta(hours=h))
        )

        # Ordre des colonnes : Date, Ech., puis membres numériques
        cols_membres = [m for m in self.MEMBRES if m in df.columns]
        df = df[["Date", "Ech."] + cols_membres].reset_index(drop=True)

        # Matrice couleurs (lignes × membres uniquement)
        colors_matrix = [
            [generer_couleur_t850(df.at[i, m]) for m in cols_membres]
            for i in df.index
        ]

        info_str = (
            f"Run GEFS {run_dt.strftime('%d/%m/%Y %H')}Z — S3 noaa-gefs-pds"
            + (f" ({len(erreurs_s3)} membre(s) en erreur)" if erreurs_s3 else "")
        )
        return df, colors_matrix, info_str

class ECMWFFetcher(WeatherAPIClient):

    def fetch_t850(self, run_dt: datetime.datetime, model: str = "ifs"):
        if ECMWFClient is None:
            return None, None, "ecmwf-opendata non installé (pip install ecmwf-opendata cfgrib)"

        try:
            client.retrieve(
                date=run_dt.strftime("%Y%m%d"),
                time=run_dt.hour,
                model=model,
                type="ep", # Ensemble Perturbed
                param="t",
                levelist=850,
                # Ne pas spécifier de 'step' ici permet de récupérer toutes les échéances disponibles
                target=target_file,
            )
# ... le reste du code ...

            ds = xr.open_dataset(target_file, engine="cfgrib")
            ds_point = ds["t"].sel(
                longitude=self.lon, latitude=self.lat, method="nearest"
            )

            df_raw = ds_point.to_dataframe().reset_index()
            df_raw["t"]          = df_raw["t"] - 273.15
            df_raw["time"]       = pd.to_datetime(df_raw["time"])
            df_raw["valid_time"] = df_raw["time"] + df_raw["step"]
            df_raw["Ech."]       = (df_raw["step"].dt.total_seconds() / 3600).astype(int)
            df_raw["Date"]       = df_raw["valid_time"].apply(formater_date_meteociel)

            df_pivot = (
                df_raw
                .pivot(index=["valid_time", "Date", "Ech."], columns="number", values="t")
                .reset_index()
            )
            df_pivot = df_pivot.sort_values("valid_time").drop(columns=["valid_time"])
            df_pivot.columns = [str(c) for c in df_pivot.columns]

            colors_matrix = [
                [generer_couleur_t850(val) for val in row]
                for _, row in df_pivot.drop(columns=["Date", "Ech."]).iterrows()
            ]

            return (
                df_pivot,
                colors_matrix,
                f"Run ECMWF {model.upper()} {run_dt.strftime('%d/%m/%Y %H')}Z",
            )

        except Exception as exc:
            return None, None, str(exc)
        finally:
            if os.path.exists(target_file):
                os.remove(target_file)

RUN_HOURS = {"0Z": 0, "12Z": 12}


def parse_file_dt(path):
    """Retourne (datetime, date, run) à partir du nom Forecast-DDMMYYYY-{0Z|12Z}.xlsx."""
    m = re.search(r"Forecast-(\d{8})(?:-(0Z|12Z))?\.xlsx", os.path.basename(path))
    if not m:
        return None
    d = datetime.strptime(m.group(1), "%d%m%Y")
    run = m.group(2) or "12Z"
    return d + timedelta(hours=RUN_HOURS[run]), d.date(), run

def parse_valid_time(s):
    """'2026-06-23 12Z' → Timestamp; gère aussi un datetime déjà parsé."""
    if isinstance(s, (pd.Timestamp, datetime)):
        return pd.Timestamp(s)
    parts = str(s).split()
    base = pd.to_datetime(parts[0], errors="coerce")
    if pd.isna(base):
        return pd.NaT
    hour = 0
    if len(parts) > 1:
        m = re.search(r"(\d{1,2})Z", parts[1])
        if m:
            hour = int(m.group(1))
    return base + pd.Timedelta(hours=hour)


def _parse_date_column(series, fallback_year):
    """Parse une colonne Date hétérogène : datetime, 'YYYY-MM-DD', ou 'jj/mm'.
    Pour le format 'jj/mm' (sans année), on complète avec fallback_year."""
    out = pd.to_datetime(series, errors="coerce", format="mixed", dayfirst=False)
    miss = out.isna()
    if miss.any():
        # Reparse les jj/mm en injectant l'année du run
        patched = series[miss].astype(str) + f"/{fallback_year}"
        out.loc[miss] = pd.to_datetime(patched, format="%d/%m/%Y", errors="coerce")
    return out

    Retourne (df_long_stats, members_df, det_series)."""
    df = _safe_read_excel(path, sheet_name=sheet, skiprows=3)

        return None
    df["valid_time"] = df["Date"].apply(parse_valid_time)

    skip = {"Date", "Ech.", "valid_time"}

        skip.add(det_col)
    member_cols = [c for c in df.columns if c not in skip]

    members = df[member_cols].apply(pd.to_numeric, errors="coerce")
    stats = pd.DataFrame({"valid_time": df["valid_time"]})
    stats["Ech"] = pd.to_numeric(df.get("Ech."), errors="coerce")
    stats["min"] = members.min(axis=1)
    stats["p10"] = members.quantile(0.10, axis=1)
    stats["p25"] = members.quantile(0.25, axis=1)
    stats["median"] = members.median(axis=1)
    stats["p75"] = members.quantile(0.75, axis=1)
    stats["p90"] = members.quantile(0.90, axis=1)
    stats["max"] = members.max(axis=1)
    stats["spread"] = (stats["p90"] - stats["p10"]).round(2)
    stats["n_members"] = members.notna().sum(axis=1)
    det = pd.to_numeric(df[det_col], errors="coerce") if det_col is not None else None
    members.index = df["valid_time"]
    return stats, members, det

        fillcolor=_rgba(color, opacity), name=name, hoverinfo="skip"))


def _rgba(hex_color, alpha):
    h = hex_color.lstrip("#")
    r, g, b = int(h[0:2], 16), int(h[2:4], 16), int(h[4:6], 16)
    return f"rgba({r},{g},{b},{alpha})"


def fan_chart(syn, title):
    """Panache de dispersion du super-ensemble : Min–Max, P10–P90, P25–P75, médiane."""
    x = syn["valid_time"]
    fig = go.Figure()
    base = "#2C3E50"
    _band(fig, x, syn["min"] if "min" in syn else syn["Min"],
          syn["max"] if "max" in syn else syn["Max"], base, "Min–Max", 0.08)
    _band(fig, x, syn.get("p10", syn.get("P10")), syn.get("p90", syn.get("P90")),
          base, "P10–P90", 0.16)
    _band(fig, x, syn.get("p25", syn.get("P25")), syn.get("p75", syn.get("P75")),
          base, "P25–P75 (50 %)", 0.28)
    med = syn.get("median", syn.get("Médiane"))
    fig.add_trace(go.Scatter(x=x, y=med, mode="lines+markers", name="Médiane",
                             line=dict(color="#E74C3C", width=3),
                             marker=dict(size=5)))
    fig.update_layout(
        title=title, height=480, hovermode="x unified",
        xaxis_title="Échéance (date de validité)", yaxis_title="Température (°C)",
        legend=dict(orientation="h", y=1.08), template="plotly_white",
        margin=dict(t=70, l=10, r=10, b=10),
    )
    return fig


def spaghetti_chart(members, stats, det, model, det_label):
    """Tous les membres d'ensemble (fins) + médiane + run déterministe."""
    fig = go.Figure()
    color = MODEL_COLORS.get(model, "#888")
    x = members.index
    for col in members.columns:
        fig.add_trace(go.Scatter(
            x=x, y=members[col], mode="lines",
            line=dict(color=_rgba(color, 0.22), width=0.8),
            hoverinfo="skip", showlegend=False))
    fig.add_trace(go.Scatter(
        x=stats["valid_time"], y=stats["median"], mode="lines",
        name="Médiane", line=dict(color=color, width=3.5)))
    if det is not None:
        fig.add_trace(go.Scatter(
            x=stats["valid_time"], y=det.values, mode="lines",
            name=f"Déterministe ({det_label})",
            line=dict(color="#111", width=2, dash="dash")))
    fig.update_layout(
        title=f"Spaghetti des membres — {model} ({members.shape[1]} scénarios)",
        height=480, hovermode="x unified", template="plotly_white",
        xaxis_title="Échéance (date de validité)", yaxis_title="Température (°C)",
        legend=dict(orientation="h", y=1.08), margin=dict(t=70, l=10, r=10, b=10),
    )
    return fig


def models_median_chart(path, sig, models):
    """Comparaison des médianes (et enveloppe P10–P90) des modèles."""
    fig = go.Figure()
    for model in models:
        loaded = load_model(path, model, sig)
        if not loaded:
            continue
        stats, _, _ = loaded
        c = MODEL_COLORS[model]
        _band(fig, stats["valid_time"], stats["p10"], stats["p90"], c,
              f"{model} P10–P90", 0.12)
        fig.add_trace(go.Scatter(
            x=stats["valid_time"], y=stats["median"], mode="lines",
            name=f"{model} médiane", line=dict(color=c, width=2.8)))
    fig.update_layout(
        title="Comparaison des modèles — médianes & dispersion",
        height=480, hovermode="x unified", template="plotly_white",
        xaxis_title="Échéance (date de validité)", yaxis_title="Température (°C)",
        legend=dict(orientation="h", y=1.1), margin=dict(t=80, l=10, r=10, b=10),
    )
    return fig


def spread_chart(syn):
    """Incertitude (spread P90-P10 et écart-type) en fonction de l'échéance."""
    fig = go.Figure()
    if "Spread" in syn:
        fig.add_trace(go.Bar(x=syn["valid_time"], y=syn["Spread"], name="Spread (P90−P10)",
                             marker_color=_rgba("#2980B9", 0.55)))
    if "Ecart-type" in syn:
        fig.add_trace(go.Scatter(x=syn["valid_time"], y=syn["Ecart-type"],
                                 name="Écart-type", yaxis="y2",
                                 line=dict(color="#C0392B", width=2.5)))
    fig.update_layout(
        title="Incertitude de la prévision selon l'échéance",
        height=360, template="plotly_white", hovermode="x unified",
        xaxis_title="Échéance", yaxis_title="Spread (°C)",
        yaxis2=dict(title="Écart-type (°C)", overlaying="y", side="right"),
        legend=dict(orientation="h", y=1.15), margin=dict(t=70, l=10, r=10, b=10),
    )
    return fig

def page_overview(runs, sig):
    st.title("🌡️ Dashboard Météo — Prévisions d'ensemble (Paris)")
    if runs.empty:
        st.warning("Aucun fichier de prévision trouvé dans le dossier `Forecasts/`.")
        return

    latest = runs.iloc[0]
    st.caption(f"Run le plus récent : **{latest['label']}** · {len(runs)} runs archivés")

    syn = load_synthese(latest["path"], sig)
    if syn is None:
        st.error("Feuille Synthèse illisible pour ce run.")
        return

    # KPI sur la première échéance « jour » exploitable
    c1, c2, c3, c4 = st.columns(4)
    med_col = "Médiane"
    today_row = syn.iloc[0]
    c1.metric("Médiane prochaine échéance", f"{today_row[med_col]:.1f} °C")
    if "Spread" in syn:
        c2.metric("Spread moyen", f"{syn['Spread'].mean():.1f} °C",
                  help="Largeur moyenne P90−P10 sur toutes les échéances")
    peak = syn.loc[syn[med_col].idxmax()]
    c3.metric("Pic médian prévu", f"{peak[med_col]:.1f} °C",
              help=f"{peak['valid_time'].strftime('%d %b')}")
    if "Proba > 20°" in syn:
        c4.metric("Jours médiane > 20°", int((syn[med_col] > 20).sum()),
                  help="Nombre d'échéances avec médiane au-dessus de 20 °C")

    st.plotly_chart(fan_chart(syn, f"Panache du super-ensemble — {latest['label']}"),
                    use_container_width=True)

    models = available_models(latest["path"])
    if models:
        st.plotly_chart(models_median_chart(latest["path"], sig, models),
                        use_container_width=True)


def page_explore(runs, sig):
    st.title("📊 Explorer un run")
    if runs.empty:
        st.warning("Aucun run disponible.")
        return

    idx = st.selectbox("Choisir un run", runs.index,
                       format_func=lambda i: runs.loc[i, "label"])
    run = runs.loc[idx]
    path = run["path"]
    st.caption(f"Fichier : `{os.path.basename(path)}`")

    syn = load_synthese(path, sig)
    models = available_models(path)

    tab_fan, tab_spag, tab_cmp, tab_unc, tab_tbl = st.tabs(
        ["📈 Panache", "🍝 Spaghetti", "⚖️ Modèles", "📉 Incertitude", "🧾 Tableaux"]
    )

    with tab_fan:
        if syn is not None:
            st.plotly_chart(fan_chart(syn, f"Super-ensemble — {run['label']}"),
                            use_container_width=True)
        else:
            st.info("Pas de feuille Synthèse dans ce run.")

    with tab_spag:
        if models:
            model = st.radio("Modèle", models, horizontal=True, key="spag_model")
            loaded = load_model(path, model, sig)
            if loaded:
                stats, members, det = loaded
                st.plotly_chart(
                    spaghetti_chart(members, stats, det, model, DET_LABEL.get(model, "DET")),
                    use_container_width=True)
        else:
            st.info("Aucune feuille modèle individuelle dans ce run.")

    with tab_cmp:
        if models:
            st.plotly_chart(models_median_chart(path, sig, models),
                            use_container_width=True)
            comp = load_comparaison(path, sig)
            if comp is not None and "Divergence Modèles" in comp.columns:
                figd = go.Figure()
                figd.add_trace(go.Bar(
                    x=comp["valid_time"], y=comp["Divergence Modèles"],
                    marker_color=_rgba("#7D3C98", 0.6), name="Divergence"))
                figd.update_layout(
                    title="Divergence inter-modèles (médiane la + chaude − la + froide)",
                    height=320, template="plotly_white",
                    xaxis_title="Échéance", yaxis_title="Écart (°C)",
                    margin=dict(t=60, l=10, r=10, b=10))
                st.plotly_chart(figd, use_container_width=True)
        else:
            st.info("Comparaison indisponible (pas de feuilles modèles).")

    with tab_unc:
        if syn is not None and ("Spread" in syn or "Ecart-type" in syn):
            st.plotly_chart(spread_chart(syn), use_container_width=True)
        else:
            st.info("Pas de données d'incertitude.")

    with tab_tbl:
        sheets = list_sheets(path)
        sheet = st.selectbox("Feuille Excel", sheets, key="tbl_sheet")
        skip = 4 if ("synth" in sheet.lower() or "comparaison" in sheet.lower()) else 3
        raw = _safe_read_excel(path, sheet_name=sheet, skiprows=skip).dropna(how="all")
        styler = raw
        # Coloration douce des colonnes de température numériques
        num_cols = raw.select_dtypes(include="number").columns
        if len(num_cols):
            styler = raw.style.background_gradient(
                cmap="RdYlBu_r", subset=list(num_cols), axis=None).format(precision=1)
        st.dataframe(styler, use_container_width=True, height=520)
        st.download_button(
            "⬇️ Télécharger la feuille (CSV)",
            raw.to_csv(index=False).encode("utf-8-sig"),
            file_name=f"{os.path.splitext(os.path.basename(path))[0]}_{sheet}.csv",
            mime="text/csv")


def page_convergence(runs, sig):
    st.title("🔄 Convergence des runs")
    st.caption(
        "Comment la prévision pour une **date cible** a évolué run après run — "
        "indicateur clé de la fiabilité (un forecast stable est plus fiable)."
    )
    if len(runs) < 2:
        st.warning("Il faut au moins 2 runs pour analyser la convergence.")
        return

    # Construit une table longue : pour chaque run, médiane/P10/P90 par date de validité
    records = []
    for _, r in runs.iterrows():
        syn = load_synthese(r["path"], sig)
        if syn is None:
            continue
        for _, row in syn.iterrows():
            records.append({
                "run_dt": r["datetime"], "run_label": r["label"],
                "target": pd.Timestamp(row["valid_time"]).normalize(),
                "median": row.get("Médiane"), "p10": row.get("P10"), "p90": row.get("P90"),
            })
    long = pd.DataFrame(records).dropna(subset=["median"])
    if long.empty:
        st.warning("Données insuffisantes.")
        return

    targets = sorted(long["target"].unique())
    default_t = [t for t in targets if t >= pd.Timestamp(datetime.now().date())][:5] or targets[:5]
    chosen = st.multiselect(
        "Dates cibles à suivre", targets, default=default_t,
        format_func=lambda t: pd.Timestamp(t).strftime("%d %b %Y"))

    if chosen:
        fig = go.Figure()
        palette = ["#E74C3C", "#2980B9", "#27AE60", "#8E44AD", "#E67E22",
                   "#16A085", "#C0392B", "#2C3E50"]
        for i, t in enumerate(sorted(chosen)):
            d = long[long["target"] == t].sort_values("run_dt")
            c = palette[i % len(palette)]
            fig.add_trace(go.Scatter(
                x=d["run_dt"], y=d["p90"], mode="lines", line=dict(width=0),
                hoverinfo="skip", showlegend=False))
            fig.add_trace(go.Scatter(
                x=d["run_dt"], y=d["p10"], mode="lines", line=dict(width=0),
                fill="tonexty", fillcolor=_rgba(c, 0.12), hoverinfo="skip",
                showlegend=False))
            fig.add_trace(go.Scatter(
                x=d["run_dt"], y=d["median"], mode="lines+markers",
                name=pd.Timestamp(t).strftime("%d %b"),
                line=dict(color=c, width=2.5)))
        fig.update_layout(
            title="Médiane prévue (et enveloppe P10–P90) par run successif",
            height=480, template="plotly_white", hovermode="x unified",
            xaxis_title="Date du run", yaxis_title="Température prévue (°C)",
            legend=dict(orientation="h", y=1.08), margin=dict(t=70, l=10, r=10, b=10))
        st.plotly_chart(fig, use_container_width=True)

    # Heatmap d'évolution : lignes = date cible, colonnes = run
    st.subheader("Carte d'évolution des médianes")
    pivot = long.pivot_table(index="target", columns="run_dt", values="median")
    pivot = pivot.sort_index()
    if not pivot.empty:
        heat = go.Figure(data=go.Heatmap(
            z=pivot.values,
            x=[pd.Timestamp(c).strftime("%d %b %Hh") for c in pivot.columns],
            y=[pd.Timestamp(i).strftime("%d %b") for i in pivot.index],
            colorscale="RdYlBu_r", colorbar=dict(title="°C"),
            hovertemplate="Run %{x}<br>Cible %{y}<br>%{z:.1f} °C<extra></extra>"))
        heat.update_layout(
            height=max(300, 26 * len(pivot.index) + 120), template="plotly_white",
            xaxis_title="Run", yaxis_title="Date cible", margin=dict(t=30, l=10, r=10, b=10))
        st.plotly_chart(heat, use_container_width=True)
        st.caption("Une ligne de couleur homogène = prévision stable d'un run à l'autre.")


def page_run(sig):
    st.title("🚀 Lancer un nouveau run")
    st.write(
        "Exécute `Forecast.py` : télécharge les tableaux Meteociel (ECMWF/AIFS/GEFS), "
        "construit le super-ensemble et enregistre un fichier dans `Forecasts/`."
    )
    run_choice = st.radio("Run à récupérer", ["0Z", "12Z"], horizontal=True)
    st.info("Le téléchargement nécessite une connexion internet et peut prendre ~30 s.")

    if st.button(f"▶️ Lancer le run {run_choice}", type="primary"):
        log = st.empty()
        with st.spinner(f"Exécution de Forecast.py {run_choice}…"):
            try:
                # Force l'enfant à écrire en UTF-8 (sinon ses emojis plantent en cp1252)
                child_env = {**os.environ, "PYTHONIOENCODING": "utf-8", "PYTHONUTF8": "1"}
                proc = subprocess.run(
                    [sys.executable, FORECAST_SCRIPT, run_choice],
                    cwd=BASE_DIR, capture_output=True, text=True,
                    encoding="utf-8", errors="replace", timeout=300, env=child_env,
                )
                output = (proc.stdout or "") + (("\n[stderr]\n" + proc.stderr) if proc.stderr else "")
                log.code(output or "(aucune sortie)")
                if proc.returncode == 0:
                    st.success("✅ Run terminé. Les nouvelles données sont disponibles.")
                    st.cache_data.clear()
                else:
                    st.error(f"❌ Le script s'est terminé avec le code {proc.returncode}.")
            except subprocess.TimeoutExpired:
                st.error("⏱️ Délai dépassé (5 min). Le site est peut-être inaccessible.")
            except Exception as e:  # noqa: BLE001
                st.error(f"Erreur lors du lancement : {e}")

def main():
    sig = runs_signature()
    runs = list_runs(sig)

    st.sidebar.title("🌦️ Navigation")
    page = st.sidebar.radio(
        "Aller à",
        ["Vue d'ensemble", "Explorer un run", "Convergence des runs", "Lancer un run"],
    )
    st.sidebar.markdown("---")
    st.sidebar.metric("Runs archivés", len(runs))
    if not runs.empty:
        st.sidebar.caption(f"Dernier : {runs.iloc[0]['label']}")
    if st.sidebar.button("🔄 Rafraîchir les données"):
        st.cache_data.clear()
        st.rerun()
    st.sidebar.markdown(
        "<small>Données : Meteociel · Paris · échéances 12Z</small>",
        unsafe_allow_html=True)

    if page == "Vue d'ensemble":
        page_overview(runs, sig)
    elif page == "Explorer un run":
        page_explore(runs, sig)
    elif page == "Convergence des runs":
        page_convergence(runs, sig)
    else:
        page_run(sig)


if __name__ == "__main__":
    main()
